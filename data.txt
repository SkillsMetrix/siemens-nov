import { Component } from '@angular/core';
import { FormControl, FormGroup, Validators } from '@angular/forms';

@Component({
  selector: 'app-datadriven',
  templateUrl: './datadriven.component.html',
  styleUrl: './datadriven.component.css'
})
export class DatadrivenComponent {
  myForm: FormGroup
  uname: FormControl
  em: FormControl
  pass: FormControl
  city: FormControl

  createFormControls() {
    this.uname = new FormControl('', Validators.required)
    this.pass = new FormControl('', [Validators.required, Validators.minLength(6)])
    this.em = new FormControl('', [Validators.required,Validators.email, this.emailDomainValidator])
    this.city = new FormControl('')
  }
  createForm() {
    this.myForm = new FormGroup({
      uname: this.uname,
      em: this.em,
      pass: this.pass,
      city: this.city
    })
  }
  constructor() {
    this.createFormControls()
    this.createForm()
  }
  addUser() {
    console.log(this.myForm.value);

  }
  emailDomainValidator(control: FormControl) {
     let email = control.value
    if (email && email.indexOf('@') != -1) {
      let [data, domain] = email.split('@')
      if (domain !== 'siemens.com') {
        return {
          emailDomain: {
            myDomain: domain
          }
        }
      }
    }
    return null
  }

}
